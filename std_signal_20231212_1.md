#include <csignal>
#include <iostream>

void signalHandler(int signalNumber) {
  std::cout << "Received signal: " << signalNumber << std::endl;
}

int main() {
  struct sigaction sa;
  sa.sa_handler = signalHandler;
  sigemptyset(&sa.sa_mask);
  sa.sa_flags = 0;

  // 注册 SIGINT 信号的处理程序
  if (sigaction(SIGINT, &sa, nullptr) == -1) {
    std::cerr << "Failed to register signal handler for SIGINT" << std::endl;
    return 1;
  }

  // 无限循环等待信号
  while (true) {
    // do something
  }

  return 0;
}
